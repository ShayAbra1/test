  <script>
    const wordCategories = [
      {
        category: 'אוכל ושתייה',
        words: ['sandwich', 'chocolate', 'salad', 'drink', 'juice', 'tea', 'coffee', 'chicken', 'chips', 'pizza']
      },
      {
        category: 'גוף ורגשות',
        words: ['teeth', 'nose', 'mouth', 'happy', 'disappointed', 'hungry', 'thirsty']
      },
      {
        category: 'זמנים וימים',
        words: ['Sunday', 'Monday', 'Tuesday', 'time', 'dinner', 'birthday']
      },
      {
        category: 'שפה ותקשורת',
        words: ['you', 'their', 'they', 'what', 'can', 'Mom', 'share']
      }
    ];

    let currentCategory = null;
    let currentWord = null;
    let score = 0;
    let attempts = 0;

    function startGame() {
      document.getElementById('welcome').style.display = 'none';
      document.getElementById('gameArea').style.display = 'block';
      selectRandomWord();
    }

    function selectRandomWord() {
      const randomCategory = wordCategories[Math.floor(Math.random() * wordCategories.length)];
      const randomWord = randomCategory.words[Math.floor(Math.random() * randomCategory.words.length)];
      currentCategory = randomCategory;
      currentWord = randomWord;

      document.getElementById('categoryName').innerText = currentCategory.category;
      document.getElementById('currentWord').innerText = currentWord;
      document.getElementById('translation').style.display = 'none';

      // Fetch a random image (placeholder logic used)
      document.getElementById('wordImage').src = `https://via.placeholder.com/200?text=${currentWord}`;
    }

    function checkGuess() {
      const userGuess = document.getElementById('userGuess').value.trim().toLowerCase();
      attempts++;
      if (userGuess === currentWord.toLowerCase()) {
        score++;
      }
      document.getElementById('score').innerText = score;
      document.getElementById('attempts').innerText = attempts;
      document.getElementById('userGuess').value = '';
      selectRandomWord();
    }

    function showTranslation() {
      document.getElementById('translation').style.display = 'block';
    }
  </script>
